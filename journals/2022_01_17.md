	- 需要维护一个内置的 buffer
	- 需要支持 Read & Seek
	- 考虑模仿一下 tokio 对 file 的实现
		- ```rust
		  #[derive(Debug)]
		  pub(crate) struct Blocking<T> {
		      inner: Option<T>,
		      state: State<T>,
		      /// `true` if the lower IO layer needs flushing.
		      need_flush: bool,
		  }
		  
		  #[derive(Debug)]
		  pub(crate) struct Buf {
		      buf: Vec<u8>,
		      pos: usize,
		  }
		  
		  
		  #[derive(Debug)]
		  enum State<T> {
		      Idle(Option<Buf>),
		      Busy(sys::Blocking<(io::Result<usize>, Buf, T)>),
		  }
		  
		  ```
